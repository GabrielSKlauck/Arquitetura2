INCLUDE "emu8086.inc"

TAMANHO EQU 32

org 100h

DEFINE_GET_STRING
DEFINE_PRINT_STRING

MOV BX, 0
JMP INICIO

INICIO:
    MOV AH, 0
    MOV AL, 00h
    INT 10H
    PUSH BX
    MOV CONTADOR_LINHA, 1
    MOV CONTADOR_COLUNA, 0    
    MOV AH, 2    
    MOV DH, CONTADOR_LINHA
    MOV DL, CONTADOR_COLUNA 
    MOV BH, 0
    INT 10H
    LEA SI, MSG_1
    CALL PRINT_STRING
    POP BX
    PUSH BX
    MOV SI,PONTEIRO_PALAVRAS[BX]
    CALL PRINT_STRING
    CALL RESET_MOUSE
    MOV CX, 0
    MOV SI, OFFSET INICIO_A
    CALL ESCREVER_LABIRINTO
    
   
PROCURAR_PALAVRA:
    POP BX
    MOV SI,PONTEIRO_PALAVRAS[BX]
    MOV DI, OFFSET INICIO_A
                                                              
    CALL RESET_MOUSE                                  
    CALL PROCURA_ESQUERDA_CIMA
    
    
    MOV SI,PONTEIRO_PALAVRAS[BX]
    MOV DI, OFFSET INICIO_A                                 
                                 
    CALL RESET_MOUSE
    CALL PROCURA_DIREITA_CIMA
    
    
    MOV SI,PONTEIRO_PALAVRAS[BX]
    MOV DI, OFFSET INICIO_A                                 
                                 
    CALL RESET_MOUSE
    CALL PROCURA_ESQUERDA_BAIXO
    
    
    MOV SI,PONTEIRO_PALAVRAS[BX]
    MOV DI, OFFSET INICIO_A
    
    CALL RESET_MOUSE
    CALL PROCURA_DIREITA_BAIXO
    
    
    MOV SI,PONTEIRO_PALAVRAS[BX]
    MOV DI, OFFSET INICIO_A
    
    CALL RESET_MOUSE
    CALL PROCURA_ESQUERDA_DIREITA
    
    
    MOV SI,PONTEIRO_PALAVRAS[BX]
    MOV DI, OFFSET INICIO_A
    
    CALL RESET_MOUSE
    CALL PROCURA_DIREITA_ESQUERDA
    
    
    MOV SI,PONTEIRO_PALAVRAS[BX]
    MOV DI, OFFSET INICIO_A
    
    CALL RESET_MOUSE
    CALL PROCURA_CIMA_BAIXO
    
    
    MOV SI,PONTEIRO_PALAVRAS[BX]
    MOV DI, OFFSET INICIO_A
    
    CALL RESET_MOUSE
    CALL PROCURA_BAIXO_CIMA
    
    
    MOV AH, 01H
    INT 21H
    
    
    INC BX
    PUSH BX
    MOV SI,PONTEIRO_PALAVRAS[BX]
    MOV DL,[SI]                             
    CMP DL, 0
    JE FIM_PROCURAR_PALAVRA
    JMP PROCURAR_PALAVRA 
    
    
FIM_PROCURAR_PALAVRA:
    INC BX
    MOV SI,PONTEIRO_PALAVRAS[BX]
    MOV DL,[SI]
    CMP DL, 0
    JNE INICIO
    MOV AH,4CH
    MOV AL, 0
    INT 21H 
              
           
PROCURA_ESQUERDA_DIREITA:;-------------------------------------------------------------------
    PUSHF  
    PUSH DX
    PUSH DI 
    
PROCURANDO_ESQUERDA_DIREITA_IGUAIS:
    MOV DL, [SI]
    MOV DH, [DI]
    CMP DL, 0
    JE  INICIO_ACHOU_SUBSTRING_PROCURA_ESQUERDA_DIREITA
    CMP DH, 0
    JE  NAO_ACHOU_SUBSTRING_PROCURA_ESQUERDA_DIREITA
    CMP DH,DL
    JE IGUAL_PARCIAL_PROCURANDO_ESQUERDA_DIREITA_IGUAIS 
    MOV SI,PONTEIRO_PALAVRAS[BX]
    POP DI
    INC DI
    INC CONTADOR_COLUNA
    PUSH DI
    CMP CONTADOR_COLUNA, 22
    JE ADD_LINHA_ESQUERDA_DIREITA
    JMP PROCURANDO_ESQUERDA_DIREITA_IGUAIS
    
    
ADD_LINHA_ESQUERDA_DIREITA:
    MOV CONTADOR_COLUNA, 0
    INC CONTADOR_LINHA
    JMP PROCURANDO_ESQUERDA_DIREITA_IGUAIS

    
IGUAL_PARCIAL_PROCURANDO_ESQUERDA_DIREITA_IGUAIS:
    INC SI
    INC DI
    JMP PROCURANDO_ESQUERDA_DIREITA_IGUAIS
    
INICIO_ACHOU_SUBSTRING_PROCURA_ESQUERDA_DIREITA:

    MOV DH, CONTADOR_LINHA
    MOV DL, CONTADOR_COLUNA
    MOV CONTADOR_LINHA_TEMPORARIO, DH
    MOV CONTADOR_COLUNA_TEMPORARIO, DL
    JMP ACHOU_SUBSTRING_PROCURA_ESQUERDA_DIREITA
    
ACHOU_SUBSTRING_PROCURA_ESQUERDA_DIREITA:
    PUSH BX
    MOV AH, 02H
    MOV DH, CONTADOR_LINHA
    MOV DL, CONTADOR_COLUNA
    MOV BH, 0
    INT 10H
    MOV AH, 09H
    MOV AL, [SI]
    MOV BH, 0
    MOV BL, 1111B
    INT 10H
    INC SI
    INC CONTADOR_COLUNA
    MOV AL, [SI]
    POP BX
    CMP AL, 0
    JE FIM_ACHOU_SUBSTRING_PROCURA_ESQUERDA_DIREITA
    JMP ACHOU_SUBSTRING_PROCURA_ESQUERDA_DIREITA
    
    
FIM_ACHOU_SUBSTRING_PROCURA_ESQUERDA_DIREITA:
    MOV DH, CONTADOR_LINHA_TEMPORARIO
    MOV DL, CONTADOR_COLUNA_TEMPORARIO
    MOV CONTADOR_LINHA, DH
    MOV CONTADOR_COLUNA, DL

NAO_ACHOU_SUBSTRING_PROCURA_ESQUERDA_DIREITA:

FIM_PROCURA_ESQUERDA_DIREITA:
    POP DI
    POP DX
    POPF
    RET

    MOV DL, CONTADOR_COLUNA
    MOV CONTADOR_LINHA_TEMPORARIO, DH
    MOV CONTADOR_COLUNA_TEMPORARIO, DL
    JMP ACHOU_SUBSTRING_PROCURA_DIREITA_BAIXO
    
ACHOU_SUBSTRING_PROCURA_DIREITA_BAIXO:
    PUSH BX
    MOV AH, 02H
    MOV DH, CONTADOR_LINHA
    MOV DL, CONTADOR_COLUNA
    MOV BH, 0
    INT 10H
    MOV AH, 09H
    MOV AL, [SI]
    MOV BH, 0
    MOV BL, 1111B
    INT 10H
    INC SI
    DEC CONTADOR_COLUNA
    DEC CONTADOR_LINHA
    MOV AL, [SI]
    POP BX
    CMP AL, 0
    JE FIM_ACHOU_SUBSTRING_PROCURA_DIREITA_BAIXO
    JMP ACHOU_SUBSTRING_PROCURA_DIREITA_BAIXO    
    
FIM_ACHOU_SUBSTRING_PROCURA_DIREITA_BAIXO:
    MOV DH, CONTADOR_LINHA_TEMPORARIO
    MOV DL, CONTADOR_COLUNA_TEMPORARIO
    MOV CONTADOR_LINHA, DH
    MOV CONTADOR_COLUNA, DL
    MOV SI,PONTEIRO_PALAVRAS[BX]
    JMP PROCURANDO_DIREITA_BAIXO_IGUAIS
    
NAO_ACHOU_SUBSTRING_PROCURA_DIREITA_BAIXO:
    JMP FIM_PROCURA_DIREITA_BAIXO
    
FIM_PROCURA_DIREITA_BAIXO:
    POP DI
    POP DX
    POPF
    RET



RESET_MOUSE:;---------------------------------------------------
    MOV CONTADOR_LINHA, 3
    MOV CONTADOR_COLUNA, 0    
    MOV AH, 2    
    MOV DH, CONTADOR_LINHA
    MOV DL, CONTADOR_COLUNA 
    MOV BH, 0
    INT 10H
    RET


ESCREVER_LABIRINTO:           
    MOV AL, [SI]
    MOV AH, 9
    MOV BL, 1110b
    MOV CX, 1
    INT 10H
    INC SI
    MOV AL, [SI]
    CMP AL, 0
    JE RETURN
    JMP INICIO_SET_CURSOR_LABIRINTO
    

INICIO_SET_CURSOR_LABIRINTO:
    ADD CONTADOR_COLUNA, 1
    JMP SET_CURSOR_LABIRINTO


SET_CURSOR_LABIRINTO:
    MOV AH, 2    
    CMP CONTADOR_COLUNA, 22
    JE ADD_LINHA_LABIRINTO    
    MOV DH, CONTADOR_LINHA
    MOV DL, CONTADOR_COLUNA 
    MOV BH, 0
    INT 10H
    JMP ESCREVER_LABIRINTO
    

ADD_LINHA_LABIRINTO:    
    MOV CONTADOR_COLUNA, 0
    INC CONTADOR_LINHA
    JMP SET_CURSOR_LABIRINTO    


RETURN:
    POP BX
    JMP PROCURAR_PALAVRA
